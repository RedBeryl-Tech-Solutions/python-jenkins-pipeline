-- Use -dtmapf option to define another file

-- Syntax:
   source_datatype(length, scale), target_datatype

   NUMERIC(*, 0), INTEGER

   NUMBER(10,0), INT

   NUMBER, DECIMAL

 --Number with length 38 and without scale
   NUMBER(38), BIGINT

   VARCHAR2(4000), TEXT
